AnvilKit's API is designed for ergonomic precision. Use the facade crate to pull only systems you need. Built on top of industry-standard libraries like Bevy ECS and wgpu.

code src/main.rs
content_copy 
use anvilkit::prelude::*;

fn main() {
    App::new()
        .add_plugins(DefaultPlugins)
        .add_systems(Startup, setup_forge)
        .add_systems(Update, hammer_input)
        .run();
}

fn setup_forge(mut commands: Commands) {
    commands.spawn((
        AnvilCrate::default(),
        Transform::from_xyz(0.0, 1.0, 0.0),
    ));
}
